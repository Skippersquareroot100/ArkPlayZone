Salary procedure??CREATE OR REPLACE PROCEDURE update_salary(    p_salary INT,    p_deduction INT,    p_overtime INT,    p_overtime_rate INT,    p_email VARCHAR)LANGUAGE plpgsql AS $$DECLARE    v_total_salary NUMERIC(10,2);BEGIN    v_total_salary := (p_salary - p_deduction) + (p_overtime * p_overtime_rate);    UPDATE staff    SET salary = v_total_salary,            deduction = p_deduction,             overtime = p_overtime    WHERE email = p_email;END;$$;??Financial Trigger FunctionCREATE OR REPLACE FUNCTION update_financial_trigger()RETURNS TRIGGER AS $$BEGIN    INSERT INTO staff_financial (salary_date, "netSalary", "staffStaffId")    VALUES (NOW(), NEW.salary, NEW.staff_id);    RETURN NEW;END;$$ LANGUAGE plpgsql;?FINANCIAL TRIGGER DROP TRIGGER IF EXISTS trg_update_financial ON staff;CREATE TRIGGER trg_update_financialAFTER UPDATE OF salary ON staffFOR EACH ROWEXECUTE FUNCTION update_financial_trigger();
